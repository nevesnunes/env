# beautify / format

```bash
echo '{"a":1,"b":2}' | jq
# sort keys
echo '{"b":2,"a":1}' | jq -S '.'
# sort values
echo '{"b":2,"c":1}' | jq 'to_entries|sort_by(.value)|from_entries'
# reverse sort values
echo '{"b":2,"c":1}' | jq 'to_entries|sort_by(-.value)|from_entries'
```
