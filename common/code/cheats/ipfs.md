# Setup

```bash
install_dir=~/opt
(
mkdir -p "$install_dir"
cd "$install_dir"
"$install_dir"/ipfs-update/ipfs-update fetch
)
bin=$(find "$install_dir" \
        -maxdepth 1 \
        -type f \
        -name 'ipfs-v*' | \
    tail -1)
ln -fs "$bin" "$install_dir/ipfs"
```
