jq
```bash
jq -j '
  [
    [
      paths(scalars)
      | map(
        if type == "number"
        then "[" + tostring + "]"
        else "." + .
        end
      ) | join("")
    ],
    [
      .. | select(scalars) | @json
    ]
  ]
  | transpose
  | map(join(" = ") + "\n")
  | join("") 
'

jq -r '
  tostream
  | select(length > 1)
  | (
    .[0] | map(
      if type == "number"
      then "[" + tostring + "]"
      else "." + .
      end
    ) | join("")
  ) + " = " + (.[1] | @json)
'
```
